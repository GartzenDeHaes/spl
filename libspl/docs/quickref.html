1. Hello World Console Mode (with network support)

#include <spl/Debug.h>
#include <spl/Exception.h>
#include <spl/net/Socket.h>
#include <spl/UnitTest.h>
 
int main(int argc, char **argv)
{
try
{
CommandLine args(argc, argv);
 
// use args.HasSwitch("argname") and args.GetSwtich("argname") to get the arguments
if ( args.HasSwitch("test") )
{
// this will run spl's unit tests TestHarness::RunTests();
Log::WriteEndOfRunTotal();
return 0;
}
 
// Set logging to go to a file (prints on the console by default)
Log::Init("logfile.txt", CLOG_INFO);
 
// Main program code here
}
catch ( Exception *ex )
{
Log::WriteError( ex );
delete ex;
}
catch ( OutOfMemoryException mex )
{
Log::WriteError( &ex );
}
DEBUG_DUMP_MEM_LEAKS();
DEBUG_TEAR_DOWN(true);
Socket::ShutdownService();
return 0;
}
